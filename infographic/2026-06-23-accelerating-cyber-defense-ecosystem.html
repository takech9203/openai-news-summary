💻 コードサンプル: SIEM 統合 (GPT-5.5-Cyber)

SIEM システムからのセキュリティイベントを GPT-5.5-Cyber で分析し、脅威の優先度判定とインシデント対応の自動化を実現する統合例。

siem-integration-gpt55cyber.py 
from openai import OpenAI

client = OpenAI()

# SIEM 統合: GPT-5.5-Cyber によるセキュリティイベント分析
def analyze_siem_events(events: list[dict]) -> dict:
    """SIEM から取得したイベントを GPT-5.5-Cyber で分析する"""

    response = client.chat.completions.create(
        model="gpt-5.5-cyber",
        messages=[
            {
                "role": "system",
                "content": (
                    "You are an expert SOC analyst integrated with a SIEM platform. "
                    "Analyze security events, correlate indicators, prioritize threats "
                    "by severity, and recommend automated response actions compatible "
                    "with SOAR playbooks. Output in structured JSON with STIX 2.1 "
                    "indicators where applicable."
                )
            },
            {
                "role": "user",
                "content": (
                    f"Analyze the following SIEM events and provide:\n"
                    f"1. Threat priority classification (Critical/High/Medium/Low)\n"
                    f"2. Attack chain correlation\n"
                    f"3. Recommended SOAR playbook actions\n"
                    f"4. IoC extraction in STIX 2.1 format\n\n"
                    f"Events:\n{events}"
                )
            }
        ],
        max_tokens=4096,
        response_format={"type": "json_object"}
    )

    return response.choices[0].message.content


# 使用例: SIEM イベントの分析
siem_events = [
    {
        "timestamp": "2026-06-23T14:30:00Z",
        "source": "firewall",
        "event_type": "connection_attempt",
        "src_ip": "203.0.113.42",
        "dst_port": 445,
        "action": "blocked",
        "count": 1500
    },
    {
        "timestamp": "2026-06-23T14:31:00Z",
        "source": "edr",
        "event_type": "suspicious_process",
        "host": "workstation-107",
        "process": "powershell.exe",
        "command": "encoded_command_detected"
    }
]

result = analyze_siem_events(siem_events)
print(result)
