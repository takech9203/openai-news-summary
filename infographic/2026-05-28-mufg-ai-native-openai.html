Python ChatGPT Enterprise API - 金融分析 
from openai import OpenAI

# ChatGPT Enterprise 接続 (MUFG 専用エンドポイント)
client = OpenAI(
    api_key="sk-enterprise-mufg-...",
    organization="org-mufg-financial"
)

# 融資リスク分析
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {
            "role": "system",
            "content": (
                "あなたは MUFG の融資分析 AI です。"
                "金融規制を遵守し、データガバナンスに従って分析を行います。"
            )
        },
        {
            "role": "user",
            "content": (
                "以下の企業財務データに基づき、"
                "融資リスクスコアを算出してください。\n\n"
                "売上高: 500 億円\n"
                "営業利益率: 8.5%\n"
                "自己資本比率: 42%\n"
                "負債比率: 1.2"
            )
        }
    ],
    temperature=0.1,
    max_tokens=2000
)

# リスクスコアと分析結果を取得
analysis = response.choices[0].message.content
print(f"融資リスク分析結果:\n{analysis}")
