コードサンプル

Python GPT-5.5 Instant ヘルスケア問い合わせ例 
from openai import OpenAI

client = OpenAI()

response = client.chat.completions.create(
    model="gpt-5.5-instant",
    messages=[
        {"role": "system", "content": "あなたは医療情報を提供するアシスタントです。"},
        {"role": "user", "content": "頭痛が3日間続いています。考えられる原因は？"}
    ],
    temperature=0.3,
)
print(response.choices[0].message.content)
