コードサンプル
Python (boto3) - OpenAI GPT-5.5 via Amazon Bedrock
import boto3
import json

bedrock_runtime = boto3.client(
    "bedrock-runtime",
    region_name="us-east-1"
)

# OpenAI GPT-5.5 via Amazon Bedrock
response = bedrock_runtime.invoke_model(
    modelId="openai.gpt-5-5",
    contentType="application/json",
    accept="application/json",
    body=json.dumps({
        "messages": [
            {"role": "user", "content": "Design a serverless architecture for e-commerce."}
        ],
        "max_tokens": 4096,
        "temperature": 0.7
    })
)

result = json.loads(response["body"].read())
print(result["choices"][0]["message"]["content"])
