🧪 GLP-1 受容体作動薬の分子設計支援

from openai import OpenAI

client = OpenAI()

response = client.chat.completions.create(
    model="gpt-rosalind",
    messages=[
        {
            "role": "system",
            "content": (
                "You are a medicinal chemistry expert specializing in "
                "GLP-1 receptor agonists and metabolic disease drug discovery."
            )
        },
        {
            "role": "user",
            "content": (
                "Analyze the structure-activity relationship of semaglutide "
                "and suggest modifications to improve oral bioavailability."
            )
        }
    ],
    max_tokens=8192
)
print(response.choices[0].message.content)

🔬 ターゲット発見 (with tools)

from openai import OpenAI

client = OpenAI()

response = client.chat.completions.create(
    model="gpt-rosalind",
    messages=[
        {
            "role": "system",
            "content": "You are a target discovery specialist."
        },
        {
            "role": "user",
            "content": (
                "Identify 5 novel therapeutic targets for obesity "
                "beyond GLP-1 and GIP receptors."
            )
        }
    ],
    tools=[{
        "type": "function",
        "function": {
            "name": "search_pubmed",
            "description": "Search PubMed for literature",
            "parameters": {
                "type": "object",
                "properties": {
                    "query": {"type": "string"}
                },
                "required": ["query"]
            }
        }
    }],
    tool_choice="auto",
    max_tokens=8192
)
print(response.choices[0].message.content)
